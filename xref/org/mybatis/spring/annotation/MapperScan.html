18  import java.lang.annotation.Annotation;
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;

25  import org.mybatis.spring.mapper.MapperFactoryBean;
26  import org.mybatis.spring.mapper.MapperScannerConfigurer;
27  import org.springframework.beans.factory.support.BeanNameGenerator;
28  import org.springframework.context.annotation.Import;

30  /**
31   * Use this annotation to register MyBatis mapper interfaces when using Java
32   * Config. It performs when same work as {@link MapperScannerConfigurer} via
33   * {@link MapperScannerRegistrar}.
34   *
35   * <p>Configuration example:</p>
36   * <pre class="code">
37   * &#064;Configuration
38   * &#064;MapperScan("org.mybatis.spring.sample.mapper")
39   * public class AppConfig {
40   *
41   *   &#064;Bean
42   *   public DataSource dataSource() {
43   *     return new EmbeddedDatabaseBuilder()
44   *              .addScript("schema.sql")
45   *              .build();
46   *   }
47   *
48   *   &#064;Bean
49   *   public DataSourceTransactionManager transactionManager() {
50   *     return new DataSourceTransactionManager(dataSource());
51   *   }
52   *
53   *   &#064;Bean
54   *   public SqlSessionFactory sqlSessionFactory() throws Exception {
55   *     SqlSessionFactoryBean sessionFactory = new SqlSessionFactoryBean();
56   *     sessionFactory.setDataSource(dataSource());
57   *     return sessionFactory.getObject();
58   *   }
59   * }
60   * </pre>
61   *
62   * @since 1.2.0
63   * @see MapperScannerRegistrar
64   * @see MapperFactoryBean
65   * @version $Id$
66   */
67  @Retention(RetentionPolicy.RUNTIME)
68  @Target(ElementType.TYPE)
69  @Documented
70  @Import(MapperScannerRegistrar.class)
71  public @interface MapperScan {
72  
73    /**
74     * Alias for the {@link #basePackages()} attribute. Allows for more concise
75     * annotation declarations e.g.:
76     * {@code @EnableMyBatisMapperScanner("org.my.pkg")} instead of {@code
77     * @EnableMyBatisMapperScanner(basePackages= "org.my.pkg"})}.
78     */
79    String[] value() default {};
80  
81    /**
82     * Base packages to scan for MyBatis interfaces. Note that only interfaces
83     * with at least one method will be registered; concrete classes will be
84     * ignored.
85     */
86    String[] basePackages() default {};
87  
88    /**
89     * Type-safe alternative to {@link #basePackages()} for specifying the packages
90     * to scan for annotated components. The package of each class specified will be scanned.
91     * <p>Consider creating a special no-op marker class or interface in each package
92     * that serves no purpose other than being referenced by this attribute.
93     */
94    Class<?>[] basePackageClasses() default {};
95  
96    /**
97     * The {@link BeanNameGenerator} class to be used for naming detected components
98     * within the Spring container.
99     */
100   Class<? extends BeanNameGenerator> nameGenerator() default BeanNameGenerator.class;
101   
102   /**
103    * This property specifies the annotation that the scanner will search for.
104    * <p>
105    * The scanner will register all interfaces in the base package that also have
106    * the specified annotation.
107    * <p>
108    * Note this can be combined with markerInterface.
109    */
110   Class<? extends Annotation> annotationClass() default Annotation.class;
111 
112   /**
113    * This property specifies the parent that the scanner will search for.
114    * <p>
115    * The scanner will register all interfaces in the base package that also have
116    * the specified interface class as a parent.
117    * <p>
118    * Note this can be combined with annotationClass.
119    */
120   Class<?> markerInterface() default Class.class;
121 
122   /**
123    * Specifies which {@code SqlSessionTemplate} to use in the case that there is
124    * more than one in the spring context. Usually this is only needed when you
125    * have more than one datasource.
126    */
127   String sqlSessionTemplateRef() default "";
128 
129   /**
130    * Specifies which {@code SqlSessionFactory} to use in the case that there is
131    * more than one in the spring context. Usually this is only needed when you
132    * have more than one datasource.
133    */
134   String sqlSessionFactoryRef() default "";
135 
136 }
