16  package org.mybatis.spring.config;
17  
18  import java.lang.annotation.Annotation;
19  
20  import org.mybatis.spring.mapper.MapperFactoryBean;
21  import org.mybatis.spring.mapper.ClassPathMapperScanner;
22  import org.springframework.beans.BeanUtils;
23  import org.springframework.beans.factory.config.BeanDefinition;
24  import org.springframework.beans.factory.support.BeanNameGenerator;
25  import org.springframework.beans.factory.xml.BeanDefinitionParser;
26  import org.springframework.beans.factory.xml.ParserContext;
27  import org.springframework.beans.factory.xml.XmlReaderContext;
28  import org.springframework.context.ConfigurableApplicationContext;
29  import org.springframework.util.StringUtils;
30  import org.w3c.dom.Element;
31  
32  /**
33   * A {#code BeanDefinitionParser} that handles the element scan of the MyBatis.
34   * namespace
35   * 
36   * @author Lishu Luo
37   * @author Eduardo Macarron
38   *
39   * @since 1.2.0
40   * @see MapperFactoryBean
41   * @see ClassPathMapperScanner
42   * @version $Id$
43   */
44  
45  public class MapperScannerBeanDefinitionParser implements BeanDefinitionParser {
46  
47    private static String ATTRIBUTE_BASE_PACKAGE = "base-package";
48    private static String ATTRIBUTE_ANNOTATION = "annotation";
49    private static String ATTRIBUTE_MARKER_INTERFACE = "marker-interface";
50    private static String ATTRIBUTE_NAME_GENERATOR = "name-generator";
51    private static String ATTRIBUTE_TEMPLATE_REF = "template-ref";
52    private static String ATTRIBUTE_FACTORY_REF = "factory-ref";
53  
54    /**
55     * {@inheritDoc}
56     */
57    public synchronized BeanDefinition parse(Element element, ParserContext parserContext) {
58      ClassPathMapperScanner scanner = new ClassPathMapperScanner(parserContext.getRegistry());
59      ClassLoader classLoader = scanner.getResourceLoader().getClassLoader();
60      XmlReaderContext readerContext = parserContext.getReaderContext();
61      scanner.setResourceLoader(readerContext.getResourceLoader());
62      try {
63        String annotationClassName = element.getAttribute(ATTRIBUTE_ANNOTATION);
64        if (StringUtils.hasText(annotationClassName)) {
65          @SuppressWarnings("unchecked")
66          Class<? extends Annotation> markerInterface = (Class<? extends Annotation>) classLoader.loadClass(annotationClassName);
67          scanner.setAnnotationClass(markerInterface);
68        }
69        String markerInterfaceClassName = element.getAttribute(ATTRIBUTE_MARKER_INTERFACE);
70        if (StringUtils.hasText(markerInterfaceClassName)) {
71          Class<?> markerInterface = classLoader.loadClass(markerInterfaceClassName);
72          scanner.setMarkerInterface(markerInterface);
73        }
74        String nameGeneratorClassName = element.getAttribute(ATTRIBUTE_NAME_GENERATOR);
75        if (StringUtils.hasText(nameGeneratorClassName)) {
76          Class<?> nameGeneratorClass = classLoader.loadClass(nameGeneratorClassName);
77          BeanNameGenerator nameGenerator = BeanUtils.instantiateClass((Class<?>) nameGeneratorClass, BeanNameGenerator.class);
78          scanner.setBeanNameGenerator(nameGenerator);
79        }
80      } catch (Exception ex) {
81        readerContext.error(ex.getMessage(), readerContext.extractSource(element), ex.getCause());
82      }
83      String sqlSessionTemplateBeanName = element.getAttribute(ATTRIBUTE_TEMPLATE_REF);
84      scanner.setSqlSessionTemplateBeanName(sqlSessionTemplateBeanName);
85      String sqlSessionFactoryBeanName = element.getAttribute(ATTRIBUTE_FACTORY_REF);
86      scanner.setSqlSessionFactoryBeanName(sqlSessionFactoryBeanName);
87      scanner.registerFilters();
88      String basePackage = element.getAttribute(ATTRIBUTE_BASE_PACKAGE);
89      scanner.scan(StringUtils.tokenizeToStringArray(basePackage, ConfigurableApplicationContext.CONFIG_LOCATION_DELIMITERS));
90      return null;
91    }
