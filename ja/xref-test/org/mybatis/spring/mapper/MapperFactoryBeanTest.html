162   public void testNonSpringWithTx() throws Exception {
163     Environment original = sqlSessionFactory.getConfiguration().getEnvironment();
164 
165     MockDataSource mockDataSource = new MockDataSource();
166     mockDataSource.setupConnection(createMockConnection());
167 
168     Environment nonSpring = new Environment("non-spring", new JdbcTransactionFactory(), mockDataSource);
169     sqlSessionFactory.getConfiguration().setEnvironment(nonSpring);
170 
171     SqlSessionTemplate sqlSessionTemplate = new SqlSessionTemplate(sqlSessionFactory);
172 
173     TransactionStatus status = null;
174 
175     try {
176       status = txManager.getTransaction(new DefaultTransactionDefinition());
177 
178       find(sqlSessionTemplate);
179 
180       txManager.commit(status);
181 
182       // txManager still uses original connection
183       assertCommit();
184       assertSingleConnection();
185 
186       // SqlSessionTemplate uses its own connection
187       MockConnection mockConnection = (MockConnection) mockDataSource.getConnection();
188       assertEquals("should call commit on Connection", 1, mockConnection.getNumberCommits());
189       assertEquals("should not call rollback on Connection", 0, mockConnection.getNumberRollbacks());
190       assertCommitSession();
191     } finally {
192 
193       sqlSessionFactory.getConfiguration().setEnvironment(original);
194     }
195   }

205   private void find(SqlSessionTemplate sqlSessionTemplate, boolean addToConfig) throws Exception {
206     // recreate the mapper for each test since sqlSessionTemplate or the underlying
207     // SqlSessionFactory could change for each test
208     MapperFactoryBean<TestMapper> mapper = new MapperFactoryBean<TestMapper>();
209     mapper.setMapperInterface(TestMapper.class);
210     mapper.setSqlSessionTemplate(sqlSessionTemplate);
211     mapper.setAddToConfig(addToConfig);
212     mapper.afterPropertiesSet();
213 
214     mapper.getObject().findTest();
215   }
